### 3.2 配置管理 (config.js)

```javascript
class ConfigManager {
  constructor(mode) {
    this.mode = mode || 'default';
    this.storageKey = `lottery_config_${this.mode}`;
  }

  // 获取配置
  getConfig() {
    const data = localStorage.getItem(this.storageKey);
    return data ? JSON.parse(data) : this.getDefaultConfig();
  }

  // 保存配置
  saveConfig(config) {
    config.updatedAt = new Date().toISOString();
    localStorage.setItem(this.storageKey, JSON.stringify(config));
  }

  // 默认配置
  getDefaultConfig() {
    return {
      mode: this.mode,
      title: '幸运抽奖',
      prizes: [

        { id: 4, text: '谢谢参与', color: '#95a5a6', weight: 10 }
      ],
      blacklist: [],
      whitelist: []

}
```

### 3.3 存储管理 (storage.js)

```javascript
class StorageManager {
  // 保存配置
  static saveConfig(mode, config) {
    const key = `lottery_config_${mode}`;
    localStorage.setItem(key, JSON.stringify(config));
  }

  // 加载配置
  static loadConfig(mode) {
    const key = `lottery_config_${mode}`;
    const data = localStorage.getItem(key);
    return data ? JSON.parse(data) : null;
  }

  // 保存历史
  static saveHistory(mode, result) {
    const key = `lottery_history_${mode}`;
    const history = this.loadHistory(mode);
    history.push({
      ...result,
      timestamp: new Date().toISOString()
    });
    localStorage.setItem(key, JSON.stringify(history));
  }

  // 加载历史
  static loadHistory(mode) {
    const key = `lottery_history_${mode}`;
    const data = localStorage.getItem(key);
    return data ? JSON.parse(data) : [];
  }

  // 清除配置
  static clearConfig(mode) {
    localStorage.removeItem(`lottery_config_${mode}`);
  }
}
```

### 3.4 抽奖逻辑 (app.js)

```javascript
class LotteryApp {
  constructor() {
    this.mode = this.getModeFromUrl();
    this.configManager = new ConfigManager(this.mode);
    this.turntable = null;
    this.config = null;
  }

  // 从 URL 获取模式
  getModeFromUrl() {
    const params = new URLSearchParams(window.location.search);
    return params.get('mode') || 'default';
  }

  // 初始化
  init() {
    this.config = this.configManager.getConfig();
    this.initTurntable();
    this.bindEvents();
  }

  // 抽奖
  draw() {
    if (this.turntable.isSpinning) return;

    // 根据权重计算中奖结果
    const result = this.calculateResult();
    
    // 执行转盘动画
    this.turntable.spin(result.index, () => {
      this.showResult(result.prize);
    });
  }

  // 计算中奖结果（权重算法）
  calculateResult() {
    const prizes = this.config.prizes;
    const totalWeight = prizes.reduce((sum, p) => sum + (p.weight || 1), 0);
    let random = Math.random() * totalWeight;
    
    for (let i = 0; i < prizes.length; i++) {
      random -= prizes[i].weight || 1;
      if (random <= 0) {
        return { index: i, prize: prizes[i] };
      }
    }
    return { index: 0, prize: prizes[0] };
  }
}
```

## 6. 关键算法

### 6.1 权重抽奖算法

```javascript
function weightedRandom(prizes) {
  // 计算总权重
  const totalWeight = prizes.reduce((sum, p) => sum + (p.weight || 1), 0);
  
  // 生成随机数
  let random = Math.random() * totalWeight;
  
  // 找到中奖奖品
  for (let i = 0; i < prizes.length; i++) {
    random -= prizes[i].weight || 1;
    if (random <= 0) {
      return i;
    }
  }
  
  return 0;
}
```

### 6.2 转盘角度计算

```javascript
function calculateRotation(prizeIndex, totalPrizes) {
  // 每个奖品占的角度
  const anglePerPrize = 360 / totalPrizes;
  
  // 目标角度（从顶部开始）
  const targetAngle = prizeIndex * anglePerPrize + anglePerPrize / 2;
  
  // 多转几圈 + 目标角度
  const totalRotation = 360 * 5 + (360 - targetAngle);
  
  return totalRotation;
}
```

### 6.3 Canvas 绘制转盘

```javascript
function drawTurntable(ctx, prizes, centerX, centerY, radius) {
  const anglePerPrize = (2 * Math.PI) / prizes.length;
  
  prizes.forEach((prize, index) => {
    ctx.save();
    ctx.beginPath();
    ctx.moveTo(centerX, centerY);
    ctx.arc(centerX, centerY, radius, 
            index * anglePerPrize - Math.PI / 2, 
            (index + 1) * anglePerPrize - Math.PI / 2);
    ctx.closePath();
    
    // 填充颜色
    ctx.fillStyle = prize.color || getDefaultColor(index);
    ctx.fill();
    
    // 绘制文字
    ctx.fillStyle = '#fff';
    ctx.font = '14px Arial';
    ctx.textAlign = 'center';
    ctx.fillText(prize.text, 
                 centerX + Math.cos(angle) * radius * 0.7,
                 centerY + Math.sin(angle) * radius * 0.7);
    
    ctx.restore();
  });
}
```

## 9. 性能优化

### 9.1 加载优化
- CSS/JS 内联或合并，减少 HTTP 请求
- 图片使用 Emoji 代替，减少资源加载
- 使用 CSS Transform 实现动画，启用 GPU 加速

### 9.2 运行优化
- Canvas 绘制缓存，避免重复绘制
- 事件防抖，避免频繁触发
- 配置变更时才重绘转盘
